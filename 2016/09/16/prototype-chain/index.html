JS原型链的继承

发表于 2016-09-16   |   分类于 IT   |     |   阅读次数

一、call,apply,bind

call与apply的使用

(1)call(thisArgs [,args…])
Example：

var a = {
    name:'Hxiaomin', 
    say:function(){ 
        console.log("Hi,I'm Hxiaomin!");
    }
};

function b(name){
    console.log("I'm "+ name);
    this.say();
}

b.call(a,'test');

(2)apply(thisArgs[,args[]])
Example：

function b(x,y,z){
    console.log(x,y,z);
}

b.apply(null,[1,2,3]); // 1 2 3

apply和call的唯一区别是第二个参数的传递方式不同，apply的第二个参数必须是一个数组，而call允许传递一个参数列表。
(3)bind(thisArgs [,args…])
ES5引入bind的真正目的是为了弥补call/apply的不足，由于call/apply会对目标函数自动执行，从而导致它无法在事件绑定函数中使用，因为事件绑定函数不需要我们手动执行，它是在事件被触发时由JS内部自动执行的。而bind在实现改变函数this的同时又不会自动执行目标函数。
Example：

var obj = {name:'Hxiaomin'};

/**
* 给document添加click事件监听，并绑定onClick函数
* 通过bind方法设置onClick的this为obj，并传递参数p1,p2
*/
document.addEventListener('click',onClick.bind(obj,'p1','p2'),false);

//当点击网页时触发并执行
function onClick(a,b){
    console.log(
            this.name, 
            a, //p1
            b  //p2
    )
}

二、this,prototype,constructor

(1)this
apply和call可以改变this的指向

// 定义一个全局函数
function foo() {
    console.log(this.person);
}

// 定义一个全局变量
var person = "you";
// 自定义一个对象
var myself = {
    person: "me"
};

// 等价于window.foo();
foo.apply(window);  // "you"
// 此时foo中的this === myself
foo.apply(myself);    // "me"

(2)prototype
可用于构造函数的扩展函数，也可在Array等的原型中添加扩展方法
(3)constructor
javascript中的类型

* Number（数字）
* String（字符串）
* Boolean（布尔）
* Symbol（符号）（第六版新增）
* Object（对象）
    * Function（函数）
    * Array（数组）
    * Date（日期）
    * RegExp（正则表达式）
* Null（空）
* Undefined（未定义）

这里有个比较奇怪的现象

function Person(name) {
        this.name = name;
};
Person.prototype.getName = function() {
    return this.name;
};
var p = new Person("Hxiaomin");

console.log(p.constructor === Person);  // true
console.log(Person.prototype.constructor === Person); // true
console.log(p.constructor.prototype.constructor === Person); // true

function Person(name) {
        this.name = name;
};
Person.prototype = {
    getName: function() {
        return this.name;
    }
};
// Person.prototype.constructor = Person; 加入这行代码可以解决这个问题
var p = new Person("Hxiaomin");

console.log(p.constructor === Person);  // false
console.log(Person.prototype.constructor === Person); // false
console.log(p.constructor.prototype.constructor === Person); // false

出现这个现象有高人给出的见解是酱紫的

Person.prototype = new Object({
    getName: function() {
        return this.name;
    }
});

此时Person.prototype.constructor === Object
另外的解决方法

function Person(name) {
    this.name = name;
};
Person.prototype = {
      constructor:Person,//指定constructor
    getName: function() {
        return this.name;
    }
};

三、JS定义类的几种方法

(1)构造函数法

function Person(name) {
        this.name = name;
};
Person.prototype.getName = function() {
    return this.name;
};
var p = new Person("Hxiaomin");

(2)Object.create()

var person={
    name:"aaa",
    getName: function(){
        return this.name;
    }
}
var myperson = Object.create(person);
myperson.name="Hxiaomin";
myperson.getName();

若想将getName作为私有函数

function getName(name){
    return this.name=name;
}
var person={
    name:"aaa",
    getPerson: function(){
        return getName.call(this,this.name);
    }
}
var myperson = Object.create(person);
myperson.name="Hxiaomin";
myperson.getPerson();

ps:这里顺便提一下定义私有函数的方法
上面是一种方法，还有es6使用Symbol的方法

const bar = Symbol('bar');
const snaf = Symbol('snaf');

export default class myClass{

  // 公有方法
  foo(baz) {
    this[bar](baz);
  }

  // 私有方法
  [bar](baz) {
    return this[snaf] = baz;
  }

  // ...
};

上面代码中，bar和snaf都是Symbol值，导致第三方无法获取到它们，因此达到了私有方法和私有属性的效果。

(3)极简主义

var person={
    createPerson: function(){
        var people={};
        people.name="Hxiaomin";
        people.getName=function(){
            return this.name;
        }
        return people;
    }
}
var person1=person.createPerson();
person1.getName();

极简主义中的继承

var Animal={
        createAnimal: function(){
            var people={};
            people.name="Hxiaomin";
            people.getName=function(){
                return this.name;
            }
            return people;
        }
    };

var Person= {
　　　　creatFn: function(){
　　　　　　var person = Animal.createAnimal();
　　　　　　person.say = function(){ console.log("Hi,my name is "+this.name) };
　　　　　　return person;
　　　　}
　　};
var person1= Person.creatFn();
　　person1.say();
