ps：prototype例子

function array_max()
{
var i,max=this[0];
for(i=1;i<this.length;i++)
{
if(max<this[i])
max=this[i];
}
return max;
}

Array.prototype.max=array_max;
var x=new Array(1,2,3,4,5,6);
var y=x.max();

3.数组的方法

方法 描述 使用
concat 连接两个或更多的数组，并返回结果 var a=[1,2,3].concat(4,5)
或者 var a=[1,2,3].concat([4,5])
//[1,2,3,4,5]
join 把数组的所有元素放入一个字符串，元素通过指定的分隔符进行分隔 var a = [1,2,3,4,5].join(“,”);
//a为”1,2,3,4,5”
pop 删除并返回数组的最后一个元素 var a=[1,2,3,4,5];var b=a.pop();
//a为[1,2,3,4],b为5
push 向数组末尾添加一个或更多元素，并返回新的长度 var a=[1,2,3,4,5];
var b=a.push(6);
//a为[1,2,3,4,5,6],b为6
shift 删除并返回数组的第一个元素 var a=[1,2,3];
var b=a.shift();
//a为[2,3],b为1
unshift 向数组开头添加一个或更多元素，并返回新的长度 var a=[1,2];
var b=a.unshift(3,4);
//a为[3,4,1,2],b为4
reverse 颠倒数组中元素的顺序 var a=[1,2,3];
var b=a.reverse();
// a和b都为[3,2,1]
slice 从某个已有的数组返回选定的元素 var a=[1,2,3,4,5];
var b=a.slice(2,5);
//a为[1,2,3,4,5],b为[3,4,5]
sort 对数组的元素进行排序 var a=[3,2,1];
var b=a.sort();
//a和b都为[1,2,3]
splice 删除元素，并向数组添加新元素 splice(start,deleteCount,val1,val2,…)：从start位置开始删除deleteCount项，并从该位置起插入val1,val2,…
var a = [1,2,3,4,5];
var b = a.splice(2,2,7,8,9);
//a为[1,2,7,8,9,5] b：[3,4]
toSource 返回该对象的源代码 只有 Gecko 核心的浏览器（比如 Firefox）支持该方法，也就是说 IE、Safari、Chrome、Opera 等浏览器均不支持该方法。
toString 将数组转换为字符串，并返回结果 var a=[1,2,3];
var b=a.toString();
//a为[1,2,3],b为”1,2,3”
toLocalString 将数组转换为本地字符串，并返回结果 这个在数组还没体会到，一般都是用在Date对象的
valueOf 返回数组对象的原始值 var a=[1,2,3];
console.log(a.valueOf());
//[1,2,3]
indexOf 返回要查找的项在数组中的位置,两个参数：要查找的项和表示查找起点位置的索引（可选），查找不到为-1 var a = [1, 2, 3, 4, 5, 4, 3, 2, 1];
alert(a.indexOf(4))
//3 alert(a.indexOf(4, 4)) //5
lastIndexOf 同上 alert(a.lastIndexOf(4)) //5
alert(numbers.lastIndexOf(4, 4)) //3
every 对数组中的每一项运行给定的函数，如果该函数对每一项都返回true，则返回true var num = [1,2,3,4,5,6,7,8,9];
var everyResult = num.every(function(item, index, array)
{if(item > 2) {return true;}});
console.log(everyResult); //false
some 对数组中的每一项运行给定的函数，如果该函数对任意一项返回true，则返回true var someResult = num.some(function(item)
{if(item > 2) {return true;}});
console.log(someResult); //true
filter 利用指定的函数确定是否在返回的数组中包含某一项。 var num = [1,2,3,4,5,4,3,2,1];
var filterResult = num.filter(function(item)
{if(item > 2) {return true;}});
console.log(filterResult); //[3,4,5,4,3]
map 这个数组的每一项都是在原始数组中的对应项上运行传入函数的结果 var num = [1,2,3,4,5,4,3,2,1];
var mapResult = num.map(function(item)
{return ++item;});
console.log(mapResult) //[2,3,4,5,6,5,4,3,2]
forEach 这个方法没有返回值，本质上与使用for循环迭代数组一样 var num = [1,2,3,4,5,4,3,2,1];
num.forEach(function(item)
{console.log(item);}); //依次打印数组元素
reduce 四个参数分别是，前一个值，当前值，项的索引，数组对象。然而这个函数的任何值都会作为第一个参数自动传给下一项 var num = [1,2,3,4,5];
var sum = num.reduce(function(prev, cur, index, array)
{return prev + cur;});
console.log(sum);//15
reduceRight 大部分同上，主要区别是先从右边 例子同上
Array.from 用于将两类对象(类似数组的对象和可遍历的对象)转为真正的数组
Array.of 用于将一组值，转换为数组
copyWithin 在当前数组内部，将指定位置的成员复制到其他位置（会覆盖原有成员），然后返回当前数组。Array.prototype.copyWithin(target, start = 0, end = this.length)意思是从target位置开始替换成[start，end]的内容 [1, 2, 3, 4, 5].copyWithin(0, 3) // [4, 5, 3, 4, 5]
find 用于找出第一个符合条件的数组成员。它的参数是一个回调函数，所有数组成员依次执行该回调函数，直到找出第一个返回值为true的成员，然后返回该成员。如果没有符合条件的成员，则返回undefined。
findIndex 返回第一个符合条件的数组成员的位置，如果所有成员都不符合条件，则返回-1
fill fill方法使用给定值，填充一个数组。可以接受第二个和第三个参数，用于指定填充的起始位置和结束位置。
keys 对键名的遍历
values 对键值的遍历
entries 对键值对的遍历。
includes 表示某个数组是否包含给定的值,第二个参数表示搜索的起始位置，默认为0。如果第二个参数为负数，则表示倒数的位置，如果这时它大于数组长度（比如第二个参数为-4，但数组长度为3），则会重置为从0开始。

备注：上述表格从indexOf为es5新增，从Array.from开始为es6新增内容

正则（ RegExp）

1.修饰符

修饰符 描述 使用
i 执行对大小写不敏感的匹配。
g 执行全局匹配（查找所有匹配而非在找到第一个匹配后停止）。
m 执行多行匹配。
u 点字符和Unicode字符加上u即可识别
y 叫做“粘连”（sticky）修饰符,y修饰符要求匹配必须从头部开始。

备注:修饰符u与y为es新增内容

2.方括号

表达式 描述

[^abc] 查找任何不在方括号之间的字符。
[0-9] 查找任何从 0 至 9 的数字。
[a-z] 查找任何从小写 a 到小写 z 的字符。

[adgk] 查找给定集合内的任何字符。
[^adgk] 查找给定集合外的任何字符。
(red/blue/green) 查找任何指定的选项。

?!n 匹配任何其后没有紧接指定字符串 n 的字符串。

4.尾调用优化
“尾调用优化”（Tail call optimization），即只保留内层函数的调用帧。

只有不再用到外层函数的内部变量，内层函数的调用帧才会取代外层函数的调用帧，否则就无法进行“尾调用优化”。

function addOne(a){
  var one = 1;
  function inner(b){
    return b + one;
  }
  return inner(a);
}
//上面的函数不会进行尾调用优化，因为内层函数inner用到了外层函数addOne的内部变量one。

扩展：尾递归
函数调用自身，称为递归。如果尾调用自身，就称为尾递归。
递归非常耗费内存，因为需要同时保存成千上百个调用帧，很容易发生“栈溢出”错误（stack overflow）。但对于尾递归来说，由于只存在一个调用帧，所以永远不会发生“栈溢出”错误。

function factorial(n) {
  if (n === 1) return 1;
  return n * factorial(n - 1);
}

factorial(5) // 120
